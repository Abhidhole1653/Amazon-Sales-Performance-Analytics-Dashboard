-- 1. Total Revenue
SELECT SUM(revenue) AS total_revenue FROM sales;


-- 2. Total Profit
SELECT SUM(profit) AS total_profit FROM sales;


-- 3. Total Orders
SELECT COUNT(DISTINCT order_id) AS total_orders FROM sales;


-- 4. Revenue by Year
SELECT year, SUM(revenue) AS yearly_revenue
FROM sales
GROUP BY year
ORDER BY year;


-- 5. Profit by Year
SELECT year, SUM(profit) AS yearly_profit
FROM sales
GROUP BY year
ORDER BY year;


-- 6. Revenue by Category
SELECT category, SUM(revenue) AS category_revenue
FROM sales
GROUP BY category
ORDER BY category_revenue DESC;


-- 7. Orders by Weekday vs Weekend
SELECT weekday_flag, COUNT(order_id) AS total_orders
FROM sales
GROUP BY weekday_flag;


-- 8. Top 5 Products by Revenue
SELECT product_name, SUM(revenue) AS product_revenue
FROM sales
GROUP BY product_name
ORDER BY product_revenue DESC
LIMIT 5;


-- 9. Orders by Status
SELECT order_status, COUNT(order_id) AS total_orders
FROM sales
GROUP BY order_status;


-- 10. Revenue by Country
SELECT country, SUM(revenue) AS country_revenue
FROM sales
GROUP BY country
ORDER BY country_revenue DESC;


-- 11. Revenue by State
SELECT state, SUM(revenue) AS state_revenue
FROM sales
GROUP BY state
ORDER BY state_revenue DESC;


-- 12. Customer Contribution
SELECT customer_name, SUM(revenue) AS customer_revenue
FROM sales
GROUP BY customer_name
ORDER BY customer_revenue DESC;
